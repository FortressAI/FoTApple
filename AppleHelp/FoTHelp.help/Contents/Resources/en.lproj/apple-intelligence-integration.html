📊 Implementation Details

App Intents Files

All intents are defined in packages/FoTCore/AppIntents/:

  • HealthAppIntents.swift - Personal Health (4 intents)
  • ClinicianAppIntents.swift - Clinician (3 intents)
  • EducationAppIntents.swift - Education K-18 (3 intents)
  • LegalAppIntents.swift - Legal US (4 intents)

Example Intent Implementation

import AppIntents

struct RecordHealthCheckInIntent: AppIntent {
    static var title: LocalizedStringResource = "Record Health Check-In"
    static var description = IntentDescription("Record your daily mood, symptoms, and notes")

    @Parameter(title: "Mood", description: "Your mood on a scale of 1-10")
    var mood: Int?

    @Parameter(title: "Symptoms", description: "Any symptoms you're experiencing")
    var symptoms: String?

    func perform() async throws -> some IntentResult & ProvidesDialog {
        // Generate cryptographic receipt
        let receipt = try await FoTCore.generateReceipt(
            action: "health_checkin",
            data: ["mood": mood, "symptoms": symptoms]
        )

        // Save to encrypted database
        try await HealthDatabase.save(mood: mood, symptoms: symptoms, receipt: receipt)

        return .result(dialog: "Health check-in recorded with cryptographic receipt \(receipt.id)")
    }
}
