DomainPack Protocol

All domain packs must conform to the DomainPack protocol:

public protocol DomainPack: Sendable {
    /// Domain name (e.g., "Clinician", "Protein")
    var name: String { get }

    /// Semantic version (e.g., "1.0.0")
    var version: String { get }

    /// Human-readable description
    var description: String { get }

    /// Domain-specific validation rules
    var validationRules: [ValidationRule] { get }

    /// Optional: Ontology schema (Cypher DDL)
    var ontologySchema: String? { get }

    /// Optional: Pre-defined Cypher queries
    var cypherQueries: [String: String]? { get }

    /// Initialize domain pack (load models, connect to services, etc.)
    func initialize(engine: VQbitEngine, akg: AKGService) async throws

    /// Execute domain-specific operation
    func execute(operation: String, parameters: [String: Any]) async throws -> [String: Any]
}

Domain Pack Development Guide

Step 1: Define Validation Rules

public var validationRules: [ValidationRule] {
    return [
        DomainConstraint(name: "rule_name", errorMessage: "Error message") { props in
            // Return true if valid, false otherwise
            guard let value = props["field"] as? String else { return false }
            return validateValue(value)
        }
    ]
}

Step 2: Define Ontology Schema

public var ontologySchema: String? {
    return """
    // Node labels (represent entities)
    (:NodeType1 {property1: type, property2: type})
    (:NodeType2 {property1: type, property2: type})

    // Relationship types
    ()-[:RELATIONSHIP_TYPE]->()
    ()-[:ANOTHER_RELATIONSHIP]->()

    // Example data
    CREATE (:NodeType1 {id: "example1", name: "Example"})
    """
}

Step 3: Define Cypher Queries

public var cypherQueries: [String: String]? {
    return [
        "find_by_id": """
            MATCH (n:NodeType {id: $id})
            RETURN n
        """,

        "find_related": """
            MATCH (n:NodeType {id: $id})-[r:RELATIONSHIP]->(m)
            RETURN m
        """
    ]
}

Step 5: Implement Execute Method

public func execute(operation: String, 
                    parameters: [String: Any]) async throws -> [String: Any] {
    switch operation {
    case "validate":
        let data = parameters["data"]
        let result = try await validate(data)
        return ["result": result]

    case "query":
        let queryName = parameters["query"] as! String
        let params = parameters["params"] as! [String: Any]
        guard let query = cypherQueries?[queryName] else {
            throw DomainError.unknownQuery
        }
        let result = try await akg.executeCypher(query, params: params)
        return ["result": result]

    default:
        throw DomainError.unknownOperation
    }
}

Domain Pack Guidelines

Best Practices

  1. Self-Contained: Domain packs should not depend on other domain packs
  2. Versioned: Use semantic versioning (MAJOR.MINOR.PATCH)
  3. Tested: Minimum 80% code coverage
  4. Documented: Complete API documentation with Swift-DocC
  5. Validated: All data validated against domain rules
  6. Audited: All operations logged to AKG with blockchain attestation
  7. Performant: < 100ms for typical queries

Validation Rule Design

Good:

DomainConstraint(name: "positive_value", errorMessage: "Value must be positive") { props in
    guard let value = props["value"] as? Double else { return false }
    return value > 0
}

Bad (too vague):

DomainConstraint(name: "valid", errorMessage: "Invalid") { props in
    return true // What are we validating?
}

Ontology Design

Good (clear semantics):

(:Protein {id: String, sequence: String, length: Int})
-[:FOLDS_INTO]->
(:SecondaryStructure {type: String, start: Int, end: Int})

Bad (unclear):

(:Thing {data: String})
-[:RELATED_TO]->
(:OtherThing {stuff: String})
