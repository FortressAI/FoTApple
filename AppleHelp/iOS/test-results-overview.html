How to Run Tests

Package Tests:

cd /path/to/FoTApple

# Run all package tests
swift test

# Run specific package
cd packages/FoTClinician
swift test

# Run with coverage
swift test --enable-code-coverage

iOS App Tests:

cd apps/ClinicianApp/iOS

# Automated setup
./setup.sh

# Run tests
./build.sh test

# Run with verbose output
swift test --verbose

Medical Board Exam Tests:

# Run validation suite
swift test --filter MedicalBoardExamTests

# Run specific specialty
swift test --filter testCardiology
