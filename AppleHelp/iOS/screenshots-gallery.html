Screenshot Capture Commands

iOS Simulator

# Current showcase view
xcrun simctl io <SIMULATOR_ID> screenshot path/to/screenshot.png

# Record video
xcrun simctl io <SIMULATOR_ID> recordVideo --codec=h264 video.mp4
# Press Ctrl+C to stop

macOS App

# Full window capture
screencapture -w screenshot.png

# With timer
screencapture -T 3 -w screenshot.png

Safari Extension

# Use built-in screenshot tool
Cmd+Shift+4  # Area selection
