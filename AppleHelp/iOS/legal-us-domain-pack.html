Validation Rules

1. Valid Legal Citation (Bluebook Format)

DomainConstraint(name: "valid_citation", 
                 errorMessage: "Invalid legal citation format (must follow Bluebook)") { props in
    guard let citation = props["citation"] as? String else { return false }

    // Bluebook patterns
    let patterns = [
        // Supreme Court: Case v. Case, Volume U.S. Page (Year)
        #"^.+\s+v\.\s+.+,\s+\d{1,3}\s+U\.S\.\s+\d{1,4}\s+\(\d{4}\)$"#,

        // Federal statute: Title U.S.C. ¬ß Section (Year)
        #"^\d{1,2}\s+U\.S\.C\.\s+¬ß\s*\d+(\([a-z]\))?(\(\d+\))?\s+\(\d{4}\)$"#,

        // CFR: Title C.F.R. ¬ß Section (Year)
        #"^\d{1,2}\s+C\.F\.R\.\s+¬ß\s*\d+\.\d+\s+\(\d{4}\)$"#,

        // Constitution: U.S. Const. art./amend. Section
        #"^U\.S\.\s+Const\.\s+(art\.|amend\.)\s+[IVX]+,?\s*¬ß?\s*\d*$"#
    ]

    return patterns.contains { pattern in
        citation.range(of: pattern, options: .regularExpression) != nil
    }
}

Examples: - ‚úÖ Marbury v. Madison, 5 U.S. 137 (1803) - ‚úÖ 42 U.S.C. ¬ß 1983 (2018) - ‚úÖ 45 C.F.R. ¬ß 164.312 (2023) - ‚úÖ U.S. Const. art. I, ¬ß 8 - ‚úÖ U.S. Const. amend. I - ‚ùå Marbury vs Madison (wrong format) - ‚ùå First Amendment (not Bluebook style)

DomainConstraint(name: "constitutional_authority",
                 errorMessage: "Action lacks constitutional authority") { props in
    guard let action = props["action"] as? String,
          let authority = props["authority"] as? String else { return false }

    // Validate authority is an enumerated power

        // Article I, Section 8 (Congressional powers)
        "tax", "borrow", "commerce", "naturalization", "bankruptcy",
        "coin_money", "counterfeiting", "post_office", "copyright",
        "courts", "piracy", "war", "army", "navy", "militia",
        "district_of_columbia", "necessary_and_proper",

        "commander_in_chief", "pardon", "treaty", "appointment",
        "state_of_union", "convene_congress", "receive_ambassadors",
        "take_care", "commission_officers",

        // Article III (Judicial powers)
        "judicial_review", "jurisdiction", "trial_by_jury",

        "amendment_13_abolish_slavery",
        "amendment_14_equal_protection",
        "amendment_15_voting_rights",
        "amendment_19_womens_suffrage",
        "amendment_26_voting_age_18"

3. Oath Compliance

DomainConstraint(name: "oath_compliance",
                 errorMessage: "Action violates sworn oath") { props in
    guard let action = props["action"] as? String,
          let oath = props["oath_type"] as? String else { return false }

    switch oath {
    case "presidential":
        // "preserve, protect and defend the Constitution"
        return checkPresidentialOathCompliance(action)

    case "judicial":
        // "administer justice without respect to persons"
        return checkJudicialOathCompliance(action)

    case "military":
        // "support and defend the Constitution against all enemies"
        return checkMilitaryOathCompliance(action)

    case "naturalization":
        // "bear arms on behalf of the United States"
        return checkNaturalizationOathCompliance(action)

    default:
        return false
    }
}

func checkPresidentialOathCompliance(_ action: String) -> Bool {
    // Check if action preserves, protects, or defends Constitution
    let violations: Set<String> = [
        "ignore_supreme_court_order",
        "refuse_to_execute_law",
        "establish_religion",
        "suspend_habeas_corpus_unlawfully",
        "grant_title_of_nobility"
    ]

    return !violations.contains(action)
}

Presidential Oath (Article II, Section 1):

"I do solemnly swear (or affirm) that I will faithfully execute the Office of President of the United States, and will to the best of my ability, preserve, protect and defend the Constitution of the United States."

4. Jurisdiction (Valid Court Jurisdiction)

DomainConstraint(name: "jurisdiction",
                 errorMessage: "Court lacks jurisdiction") { props in
    guard let court = props["court"] as? String,
          let caseType = props["case_type"] as? String else { return false }

    switch court {
    case "supreme_court":
        // Article III + appellate jurisdiction
        return ["constitutional", "federal_question", "diversity", 
                "admiralty", "ambassador", "state_dispute"].contains(caseType)

    case "federal_district":
        // Federal question or diversity
        return ["federal_question", "diversity"].contains(caseType)

    case "state_court":
        // State law matters (unless preempted)
        return ["state_law", "state_constitution", "local_ordinance"].contains(caseType)

    default:
        return false
    }
}

5. Standing (Plaintiff Has Legal Standing)

DomainConstraint(name: "standing",
                 errorMessage: "Plaintiff lacks standing (no injury-in-fact)") { props in
    guard let injury = props["injury"] as? String,
          let causation = props["causation"] as? Bool,
          let redressability = props["redressability"] as? Bool else { return false }

    // Lujan v. Defenders of Wildlife (1992): Three requirements
    // 1. Injury in fact (concrete and particularized)
    let hasInjury = !injury.isEmpty

    // 2. Causation (fairly traceable to defendant's conduct)
    let hasCausation = causation

    // 3. Redressability (likely to be redressed by favorable decision)
    let hasRedressability = redressability

    return hasInjury && hasCausation && hasRedressability
}

Example:

// Valid standing
["injury": "unconstitutional_search_violated_4th_amendment",
 "causation": true,
 "redressability": true] // ‚úÖ Valid

// No standing (generalized grievance)
["injury": "government_wasting_tax_dollars",
 "causation": true,
 "redressability": false] // ‚ùå No standing

6. Precedent (Binding Precedent Identified)

DomainConstraint(name: "precedent",
                 errorMessage: "No binding precedent found") { props in
    guard let currentCourt = props["current_court"] as? String,
          let precedentCourt = props["precedent_court"] as? String else { return false }

    // Precedent hierarchy
    let bindingPrecedent: [String: Set<String>] = [
        "supreme_court": ["supreme_court"], // Only SCOTUS binds SCOTUS
        "circuit_court": ["supreme_court", "circuit_court"], // SCOTUS + own circuit
        "district_court": ["supreme_court", "circuit_court", "district_court"], // All higher
        "state_supreme": ["supreme_court", "state_supreme"], // SCOTUS + own state
        "state_appellate": ["supreme_court", "state_supreme", "state_appellate"],
        "state_trial": ["supreme_court", "state_supreme", "state_appellate", "state_trial"]
    ]

    return bindingPrecedent[currentCourt]?.contains(precedentCourt) ?? false
}

final class BarExamTests: XCTestCase {
    var akg: AKGService!
    var engine: VQbitEngine!
    var legalPack: LegalUSDomainPack!

    override func setUp() async throws {
        akg = try await AKGService(databasePath: ":memory:")
        engine = await VQbitEngine(dimension: 8096)
        legalPack = LegalUSDomainPack()
        try await legalPack.initialize(engine: engine, akg: akg)
    }

    // MBE Question 1: Commerce Clause
    func testCommerceClause() async throws {
        let question = """
        Congress enacted a statute prohibiting the possession of firearms 
        within 1,000 feet of any school. Does Congress have authority 
        under the Commerce Clause?
        """

        let result = try await legalPack.analyze(question: question)

        // Correct answer: No, under United States v. Lopez (1995)
        // Gun possession near schools is not economic activity
        XCTAssertEqual(result.answer, "no_authority")
        XCTAssertTrue(result.precedent.contains("United States v. Lopez"))
        XCTAssertEqual(result.reasoning, "not_substantially_related_to_interstate_commerce")

        // Verify blockchain proof
        XCTAssertNotNil(result.attestationID)
        let proof = try await blockchain.verify(result.attestationID)
        XCTAssertTrue(proof.valid)
    }

    // MBE Question 2: Free Speech
    func testFreeSpeech_ContentBasedRestriction() async throws {
        let question = """
        A city ordinance prohibits all political signs on private property.
        Is this ordinance constitutional under the First Amendment?
        """

        let result = try await legalPack.analyze(question: question)

        // Correct answer: No, content-based restriction fails strict scrutiny
        XCTAssertEqual(result.answer, "unconstitutional")
        XCTAssertTrue(result.reasoning.contains("content-based_restriction"))
        XCTAssertTrue(result.reasoning.contains("strict_scrutiny"))
        XCTAssertTrue(result.precedent.contains("Reed v. Town of Gilbert"))
    }

    // MBE Question 3: Standing
    func testStanding_NoInjuryInFact() async throws {
        let question = """
        A taxpayer files suit challenging federal expenditures on 
        Establishment Clause grounds. Does the taxpayer have standing?
        """

        let result = try await legalPack.analyze(question: question)

        // Correct answer: Generally no, under Flast v. Cohen exceptions are narrow
        XCTAssertEqual(result.answer, "no_standing")
        XCTAssertTrue(result.precedent.contains("Flast v. Cohen"))
        XCTAssertEqual(result.reasoning, "taxpayer_standing_limited_to_establishment_clause")
    }

    // Test 100+ MBE questions...
}

1. Constitutional Compliance Checker

func checkConstitutionalCompliance(action: String, 
                                   authority: String) async throws -> ComplianceResult {
    // 1. Verify enumerated power exists
    let hasPower = try await akg.executeCypher("""
        MATCH (power:Power {name: $authority})
        WHERE power.enumerated = true
        RETURN power
    """, params: ["authority": authority])

    guard !hasPower.isEmpty else {
        return .nonCompliant(reason: "No enumerated power: \(authority)")
    }

    // 2. Check for constitutional limitations
    let limitations = try await akg.executeCypher("""
        MATCH (power:Power {name: $authority})-[:LIMITED_BY]->(amendment:Amendment)
        RETURN amendment.number, amendment.text
    """, params: ["authority": authority])

    // 3. Check precedent
    let precedent = try await findBindingPrecedent(action: action, power: authority)

    // 4. Generate opinion
    let opinion = ComplianceResult(
        compliant: true,
        authority: authority,
        limitations: limitations,
        precedent: precedent,
        reasoning: "Action authorized by \(authority), no constitutional limitations violated"
    )

    // 5. Attest to blockchain
    let attestation = try await blockchain.attest(opinion)
    opinion.attestationID = attestation.id

    return opinion
}

func detectOathViolation(action: String, 
                         actor: String,
                         oath: String) async throws -> ViolationReport {
    // Presidential Oath: "preserve, protect and defend the Constitution"
    if oath == "presidential" {
        let violations: Set<String> = [
            "ignore_court_order",
            "refuse_execute_law",
            "suspend_habeas_corpus",
            "establish_religion",
            "abridge_free_speech"
        ]

        if violations.contains(action) {
            return ViolationReport(
                violated: true,
                oath: "Presidential Oath (Art. II, ¬ß 1)",
                action: action,
                reasoning: "Action fails to 'preserve, protect and defend the Constitution'",
                remedy: "Impeachment (Art. I, ¬ß 2-3) or 25th Amendment",
                attestationID: try await blockchain.attest(self).id
            )
        }
    }

    return ViolationReport(violated: false)
}

3. Contract Risk Analyzer

func analyzeContractRisk(contractText: String) async throws -> RiskReport {
    // 1. Parse contract (extract clauses)
    let clauses = try parseContract(contractText)

    // 2. Identify high-risk clauses
    var highRisk: [Clause] = []

    for clause in clauses {
        if clause.text.contains("unlimited liability") ||
           clause.text.contains("unilateral termination") ||
           clause.text.contains("sole discretion") {
            highRisk.append(clause)
        }
    }

    // 3. Check statute compliance
    let violations = try await checkStatuteCompliance(clauses)

    // 4. Generate risk report
    return RiskReport(
        overallRisk: highRisk.isEmpty ? "low" : "high",
        highRiskClauses: highRisk,
        statuteViolations: violations,
        recommendations: generateRecommendations(highRisk, violations),
        attestationID: try await blockchain.attest(self).id
    )
}
